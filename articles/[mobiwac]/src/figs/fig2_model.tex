% Standalone-compilable on its own (pdflatex fig2_model.tex) AND \input-able into
% the IEEE paper. The standalone wrapper runs only when \FIGINPUT is NOT defined.
%
% Paper preamble (once, shared with fig1):
%   \usepackage{tikz}
%   \usetikzlibrary{arrows.meta,positioning,fit,backgrounds,calc}
%   \def\FIGINPUT{1}
% Body:
%   \begin{figure}[t]\centering\input{figs/fig2_model.tex}
%   \caption{...}\label{fig:model}\end{figure}
%
% ARCHITECTURE (mtlnet_crossattn_dualtower + next_gru (cat) +
% next_stan_flow_dualtower (reg)), verified against src/models/.../model.py:
%   - two input streams: a check-in (semantic) window and a region (spatial) window;
%   - a PRIVATE per-task encoder on each stream;
%   - a SHARED bidirectional cross-attention stack (x2) that couples the streams
%     (the only shared parameters): K/V flow both ways;
%   - a GRU category head on the shared semantic output -> next-category;
%   - a region head that is a DUAL TOWER: a shared tower on the cross-attention
%     output PLUS a private spatial tower that runs on the RAW region window
%     (bypassing the shared stack), fused inside the head -> next-region.
% A dashed boundary encloses everything: one model, one forward pass, two outputs.
% This supersedes the older BRACIS single-shared-backbone schematic.

\ifdefined\FIGINPUT\else
  \documentclass[border=4pt]{standalone}
  \usepackage{tikz}
  \usepackage{amsmath}
  \usetikzlibrary{arrows.meta,positioning,fit,backgrounds,calc}
  \begin{document}
\fi

\begin{tikzpicture}[
    >={Stealth[length=2.2mm]},
    font=\footnotesize,
    box/.style={draw, rounded corners=2.5pt, align=center, inner sep=3pt,
                minimum height=7mm, line width=0.5pt, font=\scriptsize},
    inblk/.style={box, fill=orange!10, draw=orange!55!black, minimum width=27mm,
                  font=\scriptsize},
    enc/.style={box, fill=gray!9, draw=gray!55!black, minimum width=27mm},
    shared/.style={box, fill=blue!8, draw=blue!45!black, minimum width=66mm,
                   minimum height=14mm},
    cathead/.style={box, fill=teal!10, draw=teal!55!black, minimum width=27mm,
                    minimum height=9mm},
    reghead/.style={box, fill=teal!10, draw=teal!55!black, minimum width=33mm,
                    minimum height=12mm},
    outlbl/.style={font=\scriptsize\itshape, text=black!72},
    flow/.style={->, line width=0.7pt, draw=black!75},
    kv/.style={{Stealth[length=2mm]}-{Stealth[length=2mm]}, line width=0.9pt,
               draw=blue!50!black},
    skip/.style={->, line width=0.8pt, draw=teal!60!black, dashed},
  ]

  % ---------- Row 1: two input streams ----------
  \node[inblk] (inc) {check-in window\\\textit{(semantic stream)}};
  \node[inblk, right=14mm of inc] (inr) {region window\\\textit{(spatial stream)}};

  % ---------- Row 2: private per-task encoders ----------
  \node[enc, below=6mm of inc] (enca) {semantic encoder\\\scriptsize\textit{(private)}};
  \node[enc, below=6mm of inr] (encb) {spatial encoder\\\scriptsize\textit{(private)}};

  % ---------- Row 3: shared bidirectional cross-attention ----------
  \node[shared, below=8mm of $(enca)!0.5!(encb)$] (cross) {};
  \node[anchor=north, font=\scriptsize\bfseries, text=blue!30!black]
       at ([yshift=-1.3mm]cross.north) {shared bidirectional cross-attention ($\times 2$)};

  % ---------- Row 4: two task heads ----------
  \node[cathead, below=9mm of cross.south, anchor=north, xshift=-19mm] (cath)
       {GRU\\category head};
  \node[reghead, below=9mm of cross.south, anchor=north, xshift=19mm] (regh)
       {region head\\\scriptsize\textit{dual tower:}\\\scriptsize\textit{shared $+$ private spatial}};

  % ---------- Row 5: outputs ----------
  \node[outlbl, below=4mm of cath] (oc) {$\rightarrow$ next-category};
  \node[outlbl, below=4mm of regh] (orr) {$\rightarrow$ next-region};

  % ---------- arrows ----------
  % inputs -> private encoders
  \draw[flow] (inc) -- (enca);
  \draw[flow] (inr) -- (encb);
  % encoders -> shared cross-attention
  \draw[flow] (enca.south) -- (enca.south |- cross.north);
  \draw[flow] (encb.south) -- (encb.south |- cross.north);
  % bidirectional K/V coupling, drawn in the LOWER half of the box (below the title)
  \draw[kv] ([xshift=-19mm,yshift=-3.2mm]cross.center) -- ([xshift=19mm,yshift=-3.2mm]cross.center)
       node[pos=0.5, fill=blue!8, inner sep=1pt, font=\tiny, align=center]
       {streams exchange\\information};
  % shared -> heads
  \draw[flow] (cross.south) -- (cath.north);
  \draw[flow] (cross.south) -- (regh.north);
  % heads -> outputs
  \draw[flow] (cath.south) -- (oc.north);
  \draw[flow] (regh.south) -- (orr.north);
  % private spatial path: the RAW region window feeds the region head's private
  % tower directly, bypassing the encoder + shared cross-attention.
  \draw[skip] (inr.east) -- ++(11mm,0) |- (regh.east)
       node[pos=0.27, right, font=\tiny, text=teal!55!black, align=left]
       {raw region\\(private\\spatial path)};

  % ---------- dashed boundary: one model ----------
  \begin{scope}[on background layer]
    \node[draw, dashed, rounded corners=4pt, line width=0.7pt, draw=black!55,
          fill=black!2, inner sep=6pt,
          fit=(inc)(inr)(enca)(encb)(cross)(cath)(regh)(oc)(orr)] (model) {};
  \end{scope}
  \node[anchor=south, font=\scriptsize\bfseries, text=black!72]
        at (model.north) {one model, one forward pass, two predictions};

\end{tikzpicture}

\ifdefined\FIGINPUT\else
  \end{document}
\fi
